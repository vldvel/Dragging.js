Dragging.js

GitHub

About

Dragging.js is a jQuery plugin for adding simple dragging animation to HTML elements.

Usage

  1. Include jQuery: 
    								
<script src="http://ajax.googleapis.com/ajax/libs/jquery/3.2.1/jquery.min.js"></script>
  2. Include Dragging.js: 
    								
<script src="dist/jquery.Dragging.min.js"></script>
  3. Call the Dragging.js: 
    								
$('#element').Dragging();
								
							
						
$('#element1').Dragging();
						
					
						
$('#element2').Dragging({
	horizontal: false
});
						
					
						
$('#element3').Dragging({
	speed: 600
	rotate: true
});
						
					
						
$('#element4').Dragging({
	speed: 400
	vertical: false
});
